Пример #1 Примеры регулярных выражений

<?php
// Вернет true, если "abc" найдено где-либо в $string.
ereg("abc", $string);            

// Вернет true, если "abc" найдено в начале $string.
ereg("^abc", $string);

// Вернет true, если "abc" найдено в конце $string.
ereg("abc$", $string);

// Вернет true, если браузер клиента Netscape 2, 3 или MSIE 3.
eregi("(ozilla.[23]|MSIE.3)", $_SERVER["HTTP_USER_AGENT"]);

// Поместит три слова разделенных пробелами в $regs[1], $regs[2] и $regs[3].
ereg("([[:alnum:]]+) ([[:alnum:]]+) ([[:alnum:]]+)", $string, $regs); 

// Вставит тег <br /> в начало $string.
$string = ereg_replace("^", "<br />", $string); 

// Вставит тег <br /> в конец $string.
$string = ereg_replace("$", "<br />", $string);

// Уберет все символы новой строки из $string.
$string = ereg_replace("\n", "", $string);
?>
